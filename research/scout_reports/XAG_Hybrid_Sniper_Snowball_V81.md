//@version=5
strategy("XAG Hybrid Sniper  ", overlay=true, initial_capital=10000, default_qty_type=strategy.cash, currency=currency.USD, commission_type=strategy.commission.percent, commission_value=0.02) // ปรับค่าคอมตามโบรก Forex

// ---   ---
grp_risk = "--- การจัดการความเสี่ยง ---"
risk_pct    = input.float(5.0, "Risk Per Set (%)", minval=0.1, group=grp_risk)
leverage    = input.int(100, "Leverage (X)", group=grp_risk) // Silver มักใช้ Leverage สูงได้

grp_logic = "--- Silver Sniper Settings ---"
ema_slow    = input.int(200, "Master Trend (EMA)", group=grp_logic)
atr_mult_sl = input.float(2.5, "SL Multiplier (ทนแรงสะบัด Silver)", group=grp_logic)
tp_ratio    = input.float(3.5, "RR Ratio", group=grp_logic)

// ---   ---
ema200 = ta.ema(close, ema_slow)
atr    = ta.atr(14)

//  
is_bull_engulf = close > high  and open < close 
is_pinbar_bull = (low < low ) and (close > open) and (high - close) < (close - low) * 0.25 // ไส้ต้องยาวเด่น

is_bear_engulf = close < low  and open > close 
is_pinbar_bear = (high > high ) and (close < open) and (close - low) < (high - close) * 0.25

// ---   ---
long_trend  = close > ema200
short_trend = close < ema200

buy_trigger  = long_trend and (is_bull_engulf or is_pinbar_bull)
sell_trigger = short_trend and (is_bear_engulf or is_pinbar_bear)

// คำนวณ Lot Size (Strict 5%)
sl_dist = atr * atr_mult_sl
risk_amt = strategy.equity * (risk_pct / 100)
// สำหรับ XAGUSD ใน TradingView ต้องเช็คค่า Value ต่อจุดให้ดี
qty = risk_amt / sl_dist

// ---   ---
if buy_trigger and strategy.opentrades == 0
    strategy.entry("XAG Long", strategy.long, qty=qty)

if sell_trigger and strategy.opentrades == 0
    strategy.entry("XAG Short", strategy.short, qty=qty)

// ---   ---
if strategy.position_size > 0 and close > strategy.position_avg_price + (atr * 2.0) and strategy.opentrades < 3
    strategy.entry("Add XAG", strategy.long, qty=qty/2, comment="SNOWBALL-ADD")

if strategy.position_size < 0 and close < strategy.position_avg_price - (atr * 2.0) and strategy.opentrades < 3
    strategy.entry("Add XAG", strategy.short, qty=qty/2, comment="SNOWBALL-ADD")

// ---   ---
if strategy.position_size != 0
    strategy.exit("Exit", loss=sl_dist/syminfo.mintick, profit=(sl_dist * tp_ratio)/syminfo.mintick, trail_points=(atr*2.0)/syminfo.mintick, trail_offset=(atr*0.8)/syminfo.mintick)

// ---   ---
plot(ema200, color=color.new(color.gray, 0), linewidth=2)
